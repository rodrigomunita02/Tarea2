'use strict';

const AWS = require('aws-sdk');
const docClient = new AWS.DynamoDB.DocumentClient({region: 'us-east-1'});
exports.handler = (e, cta, callback) => {
  var params = {
    TableName : 'Cuenta'
  }; 
  docClient.scan(params, function(err, data){
      if(err){
          callback(err,null);
      }else{
          callback(null, data);
      }      
  });    
};
